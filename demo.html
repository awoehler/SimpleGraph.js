SimpleGraph.js Demo

Horizontal Bar Graph

dataset = {
	"settings": { 
		"grid_spacing": 1,
		"legend" : " Highest , High , Medium , Low , Lowest , none "
	},
	"data" : [ 
		{ "value":1 },
		{ "value":2, "attr": {"fill":"#bab", "stroke-width":0} },
		{ "value":3 },
		{ "value":4 },
		{ "value":5 },
		{ "value":6 }
	]
};
sample1 = new SimpleGraph( document.getElementById( "sample1"), dataset );
sample1.render();

Vertical Bar Graph

dataset = {
	"settings": { 
		"mode":"bar_vertical", 
		"grid_spacing": 0.5,
		"bar_attr" : { 
			"stroke-width": 3, 
			"fill": "#ababab" 
		} 
	},
	"data" : [ 
		{ "value":1 },
		{ "value":3,},
		{ "value":2 },
		{ "value":4 }
	]
};
sample2 = new SimpleGraph( document.getElementById( "sample2"), dataset );
